# Set Theory — Mathematics ka Fundamental Concept

# Agar aapne kabhi Mathematics mein Sets nahi padhe,
# To yahan se shuru karein!
# Ye file aapko basic se advanced Set Theory sikha degi.


# ═══════════════════════════════════════════════
# 1. SET KYA HOTA HAI?
# ═══════════════════════════════════════════════

# Set = Unique objects ka collection

# Simple definition:
# "Set ek aisa collection hai jisme har cheez sirf EK BAAR hoti hai."

# Jaise:
# Aapke bag mein kya hai?
# Pen, Pencil, Eraser, Sharpener

# Ye ek SET hai!
# Kyun?
# Kyunke har item unique hai — ek hi pen do baar nahi hai.


# ═══════════════════════════════════════════════
# 2. REAL LIFE EXAMPLES
# ═══════════════════════════════════════════════

# Example 1: Classroom Students

# Class ke sab students ka collection — Ye ek set hai!
# Kyunke har student unique hai.

# Example 2: Mobile Contacts

# Aapke phone ke contacts — Ye ek set hai!
# Ek hi number do baar save nahi hota.

# Example 3: Spice Box (Masala Dabba)

# Ghar mein masala dabba hota hai:
# Namak, Mirch, Haldi, Dhaniya, Garam Masala

# Ye ek SET hai!
# Har masala alag hai, koi duplicate nahi.

# Example 4: NOT a Set

# Aapki shopping bag mein:
# 2 Bread, 3 Milk, 1 Eggs

# Ye SET nahi hai (directly)!
# Kyunke Bread 2 baar hai.

# Lekin agar hum sirf ITEMS dekhein (quantity ignore karein):
# {Bread, Milk, Eggs} → Ye SET ban gaya!


# ═══════════════════════════════════════════════
# 3. SET NOTATION
# ═══════════════════════════════════════════════

# Mathematics mein set ko Curly Braces {} mein likhte hain.

# Set A = {1, 2, 3, 4, 5}
# Set B = {Apple, Banana, Mango}
# Set C = {a, e, i, o, u}   (Vowels ka set)

# Symbolically:
# A = {1, 2, 3, 4, 5}

# Padhte hain:
# "Set A contains elements 1, 2, 3, 4, 5"
# Ya: "A is a set of 1, 2, 3, 4, 5"


# Belongs To Symbol (∈)

# Agar 3 set A mein hai:
# 3 ∈ A   (3 belongs to A)

# Agar 10 set A mein nahi hai:
# 10 ∉ A   (10 does not belong to A)


# ═══════════════════════════════════════════════
# 4. PROPERTIES OF SETS
# ═══════════════════════════════════════════════

# Property 1: UNORDERED

# Set mein items ka ORDER matter nahi karta!

# A = {1, 2, 3}
# B = {3, 1, 2}

# A aur B SAME set hain!
# Kyunke dono mein same elements hain, chahe order different ho.

# Jaise:
# Aapke bag mein pen pehle rakho ya baad mein,
# items wahi hain.


# Property 2: NO DUPLICATES

# Set mein har item sirf EK BAAR aata hai!

# {1, 2, 2, 3, 3, 3} = {1, 2, 3}

# Jaise:
# Agar aap apni class ke students ki list banao,
# to ek student ka naam do baar nahi likhoge.

# Roll Number 15 — Sirf ek hi student ka hoga!


# Property 3: WELL-DEFINED

# Set ke elements CLEAR hone chahiye!

# "Beautiful flowers ka set" — Ye SET nahi hai!
# Kyun?
# Beautiful subjective hai — kisi ke liye rose beautiful,
# kisi ke liye sunflower.

# "Red flowers ka set" — Ye SET hai!
# Kyun?
# Red color clearly defined hai.


# ═══════════════════════════════════════════════
# 5. TYPES OF SETS
# ═══════════════════════════════════════════════

# --- Empty Set (Null Set) ---

# Wo set jisme koi element nahi hota.

# Symbol: ∅  ya  {}

# Example:
# 30+ age ke students in Class 5 = ∅
# (Kyunke Class 5 mein itni badi age ka koi student nahi)


# --- Singleton Set ---

# Wo set jisme sirf EK element ho.

# Example:
# {Moon} — Earth ka sirf ek moon hai.
# {Prime Minister of Pakistan} — Ek waqt mein sirf ek PM.


# --- Finite Set ---

# Wo set jisme limited number of elements hon.

# Example:
# Days in a week = {Mon, Tue, Wed, Thu, Fri, Sat, Sun}
# Vowels = {a, e, i, o, u}


# --- Infinite Set ---

# Wo set jisme unlimited elements hon.

# Example:
# Natural Numbers = {1, 2, 3, 4, ...}  (Kabhi khatam nahi hote)
# Even Numbers = {2, 4, 6, 8, ...}


# --- Equal Sets ---

# Do sets EQUAL hote hain agar unke elements same hon.

# A = {1, 2, 3}
# B = {3, 2, 1}
# A = B   ✅ (Order matter nahi karta)


# --- Subset ---

# Agar set A ke saare elements set B mein hon,
# to A subset hai B ka.

# Symbol: A ⊆ B

# Example:
# A = {1, 2}
# B = {1, 2, 3, 4, 5}

# A ⊆ B   (A, B ka subset hai)
# Kyunke A ke 1 aur 2 dono B mein hain.

# Har set APNA subset hota hai: A ⊆ A
# Empty set HAR set ka subset hota hai: ∅ ⊆ A


# --- Universal Set ---

# Wo set jisme discussion ke saare elements hon.

# Symbol: U  ya  ξ

# Example:
# Agar hum Class 5 ke students ki baat kar rahe hain,
# to Class 5 ke sab students = Universal Set

# Agar hum Vowels ki baat kar rahe hain,
# to English Alphabet = Universal Set


# ═══════════════════════════════════════════════

# Venn Diagram = Sets ko VISUALLY dikhane ka tareeqa.

# Universal Set = Rectangle

#    ┌─────────────────────────┐

#    │                         │
#    │    ┌───────────┐        │

#    │    └───────────┘        │
#    │                         │
#    │         4  5  6         │
#    │                         │
#    └─────────────────────────┘


# Two Sets: A and B

#    ┌──────────────────────────────┐
#    │        Universal Set         │
#    │                              │
#    │   ┌─────────┐  ┌─────────┐  │
#    │   │  Set A  │  │  Set B  │  │
#    │   │         │  │         │  │
#    │   │  1  2   │  │  4  5   │  │
#    │   │     3   │  │         │  │
#    │   └─────────┘  └─────────┘  │
#    │                              │
#    │             6  7             │
#    │                              │

# Yahan:
# 3 dono sets mein hai (A ∩ B — Intersection)
# 1, 2 sirf A mein hain
# 4, 5 sirf B mein hain
# 6, 7 kisi mein nahi hain (Universal mein hain)

# 7. SET OPERATIONS (Most Important!)
# ═══════════════════════════════════════════════

# Yeh wo operations hain jo Python mein directly use hote hain!

# Let's define two sets:

A = {1, 2, 3, 4, 5}
B = {4, 5, 6, 7, 8}


# --- UNION (∪) ---

# Union = Dono sets ke SAARE elements (No duplicates)

# A ∪ B = {1, 2, 3, 4, 5, 6, 7, 8}

# Venn Diagram:

#    ┌──────────────────────────────────┐
#    │         Universal Set            │
#    │                                  │
#    │   ┌─────────┐    ┌─────────┐    │
#    │   │  Set A  │    │  Set B  │    │
#    │   │         │    │         │    │
#    │   │  1  2   │    │  6  7   │    │
#    │   │     3   │    │     8   │    │
#    │   │     ┌───┴────┴───┐     │    │
#    │   │     │  4    5    │     │    │
#    │   │     └────────────┘     │    │
#    │   └─────────┘    └─────────┘    │
#    │                                  │
#    └──────────────────────────────────┘

# Union = SAB KUCH (highlighted area)

# Real Life:
# Class A ke students + Class B ke students = Sab students


# --- INTERSECTION (∩) ---

# Intersection = Sirf COMMON elements

# A ∩ B = {4, 5}

# Venn Diagram:

#    ┌──────────────────────────────────┐
#    │         Universal Set            │
#    │                                  │
#    │   ┌─────────┐    ┌─────────┐    │
#    │   │  Set A  │    │  Set B  │    │
#    │   │         │    │         │    │
#    │   │  1  2   │    │  6  7   │    │
#    │   │     3   │    │     8   │    │
#    │   │     ┌───┴────┴───┐     │    │
#    │   │     │  4    5    │ ← YEH! │
#    │   │     └────────────┘     │    │
#    │   └─────────┘    └─────────┘    │
#    │                                  │
#    └──────────────────────────────────┘

# Intersection = Sirf beech wala common part

# Real Life:
# Aapke dost + Mere dost = Common dost


# --- DIFFERENCE (-) ---

# Difference = Pehle set mein hain, doosre mein NAHI

# A - B = {1, 2, 3}
# (A mein hain lekin B mein nahi hain)

# B - A = {6, 7, 8}
# (B mein hain lekin A mein nahi hain)

# Venn Diagram (A - B):

#    ┌──────────────────────────────────┐
#    │         Universal Set            │
#    │                                  │
#    │   ┌─────────┐    ┌─────────┐    │
#    │   │  Set A  │    │  Set B  │    │
#    │   │         │    │         │    │
#    │   │  1  2   │ ←  │  6  7   │    │
#    │   │     3   │ YEH │     8   │    │
#    │   │     ┌───┴────┴───┐     │    │
#    │   │     │  4    5    │     │    │
#    │   │     └────────────┘     │    │
#    │   └─────────┘    └─────────┘    │
#    │                                  │
#    └──────────────────────────────────┘

# Difference = Sirf A ka exclusive part (B wala nahi)

# Real Life:
# Aapke dost JO MERE dost NAHI hain


# --- SYMMETRIC DIFFERENCE (△) ---

# Symmetric Difference = Jo DONO mein COMMON NAHI hain

# A △ B = {1, 2, 3, 6, 7, 8}

# Matlab: (A - B) ∪ (B - A)

# Venn Diagram:

#    ┌──────────────────────────────────┐
#    │         Universal Set            │
#    │                                  │
#    │   ┌─────────┐    ┌─────────┐    │
#    │   │  Set A  │    │  Set B  │    │
#    │   │         │    │         │    │
#    │   │  1  2   │ ←  │  6  7   │ ←  │
#    │   │     3   │ YEH │     8   │ YEH│
#    │   │     ┌───┴────┴───┐     │    │
#    │   │     │  4    5    │     │    │
#    │   │     └────────────┘     │    │
#    │   └─────────┘    └─────────┘    │
#    │                                  │
#    └──────────────────────────────────┘

# Symmetric Diff = Exclusive parts of BOTH sets

# Real Life:
# Aapke dost + Mere dost — Common dost REMOVE
