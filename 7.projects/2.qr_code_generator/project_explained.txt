# ================================================
# FUNCTION 1: auto_detect_logo()
# ================================================

def auto_detect_logo(url):
    """URL ke accordign auto logo return karega."""

    # Step 1: URL ko lowercase mein convert karo
    url_lower = url.lower()
    # Kya karta hai? → Case-insensitive matching ke liye
    # "GitHub.com" → "github.com"
    # Ab "github.com" match karega chahe user ne kaise bhi likha ho!

    # Step 2: Logos dictionary
    logos = {
        "github.com" : "logos/github_logo.png",
        "linkedin.com" : "logos/linkedin_logo.png",
        "youtube.com" : "logos/youtube_logo.png",
        "instagram.com" : "logos/instagram_logo.png",
        "facebook.com" : "logos/facebook_logo.png",
        "twitter.com" : "logos/twitter_logo.png",
        "x.com" : "logos/x_logo.png"
    }
    # Kya hai? → Domain name → Logo file path mapping
    # Dictionary kyun? → Easy to add new logos, no long if-else chain!
    # Naya logo add karna: Bas ek line dictionary mein add karo!
    # ⚠️ Logo files 'logos/' folder mein honi chahiye!

    # Step 3: Domain match check
    for domain, logo_path in logos.items():
        if domain in url_lower:
            return logo_path
    # Kya ho raha hai? → Har domain check karo
    # logos.items() → (key, value) pairs return karta hai
    # "github.com" in "https://github.com/ibrahim" → True!
    # First match milte hi return — aage check nahi karega!

    # Step 4: No match
    return None
    # Agar koi domain match nahi hua to None return
    # Baad mein download_favicon() try karega!


# ================================================
# FUNCTION 2: download_favicon()
# ================================================

def download_favicon(url):
    """Website se favicon download karega agar available hoa to."""

    try:
        # Step 1: URL se domain extract karo
        domain = urlparse(url).netloc
        # urlparse() → URL ko parts mein todta hai
        # "https://github.com/codewith-ibrahim"
        # .netloc → "github.com" (Domain name!)
        # .scheme → "https"
        # .path → "/codewith-ibrahim"

        # Step 2: Favicon URL banao
        favicon_url = f"https://{domain}/favicon.ico"
        # f-string se dynamic URL
        # Example: "https://github.com/favicon.ico"
        # ⚠️ .ico extension (Windows icon format)
        # Alternative: Google's favicon service (Better!)

        # Step 3: Download karo
        response = requests.get(favicon_url, timeout=5)
        # requests.get() → HTTP GET request bhejo
        # timeout=5 → 5 seconds wait, phir error!
        # response.status_code → 200 (OK), 404 (Not Found)

        # Step 4: Success check + Image convert
        if response.status_code == 200:
            favicon = Image.open(BytesIO(response.content))
            return favicon.convert("RGBA")
        # BytesIO(response.content) → Bytes ko file-like object banata hai
        # Image.open() → File-like object se image load
        # .convert("RGBA") → Transparency support ke liye

    except Exception:
        pass  # Koi bhi error → Chup chaap None return

    return None
    # Download fail → None return
    # QR Code bina logo ke generate hoga!


# ================================================
# FUNCTION 3: generate_qr() — MAIN FUNCTION
# ================================================

def generate_qr(url, logo_path=None, fill_color="black",
                back_color="white", save_as="qr_code.png"):
    """
    QR Code generate karo with logo.

    Parameters:
    url = QR Code mein encode hone wala link
    logo_path = Logo image ka path (None = auto-detect)
    fill_color = QR Code ka color (Default: black)
    back_color = Background color (Default: white)
    save_as = File name to save (Default: qr_code.png)
    """

    # --- Step 1: Logo Decide Karo (Auto ya Manual) ---
    if logo_path is None:
        logo_path = auto_detect_logo(url)
    # Agar user ne logo_path nahi diya (None) → Auto-detect!
    # auto_detect_logo() URL dekhkar logo dhunde ga
    # Mil gaya → logo_path mein file path
    # Nahi mila → logo_path = None

    # --- Step 2: QR Code Object Create ---
    qr = qrcode.QRCode(
        version=1,
        error_correction=qrcode.constants.ERROR_CORRECT_H,
        box_size=10,
        border=4,
    )
    # version=1 → Small QR (21x21), URLs ke liye perfect
    # ERROR_CORRECT_H → 30% damage recover! Logo lagane ke liye ZAROORI!
    # box_size=10 → Har module 10 pixels
    # border=4 → 4 modules ka white border (Standard)

    # --- Step 3: Data Add + Generate ---
    qr.add_data(url)
    qr.make(fit=True)
    # add_data() → URL/text QR mein encode karo
    # make(fit=True) → Auto version adjust (agar data zyada ho)

    # --- Step 4: QR Image Banao ---
    img = qr.make_image(
        fill_color=fill_color,
        back_color=back_color
    ).convert("RGBA")
    # make_image() → QR code ko image mein convert
    # .convert("RGBA") → Transparency channel add (Logo paste ke liye)

    # --- Step 5: Logo Paste Karo (Agar Available) ---
    if logo_path and os.path.exists(logo_path):
        # PATH MILA + FILE EXISTS
        logo = Image.open(logo_path).convert("RGBA")

        qr_width, qr_height = img.size
        logo_size = qr_width // 5  # Logo = 20% of QR size

        # Logo resize
        logo = logo.resize((logo_size, logo_size), Image.Resampling.LANCZOS)

        # White circle behind logo (Clean look!)
        circle_size = logo_size + 10
        circle = Image.new("RGBA", (circle_size, circle_size), (0, 0, 0, 0))
        draw = ImageDraw.Draw(circle)
        draw.ellipse((0, 0, circle_size, circle_size), fill=(255, 255, 255, 255))

        # Center calculations
        circle_x = (qr_width - circle_size) // 2
        circle_y = (qr_height - circle_size) // 2
        logo_x = (qr_width - logo_size) // 2
        logo_y = (qr_height - logo_size) // 2

        # Paste
        img.paste(circle, (circle_x, circle_y), circle)
        img.paste(logo, (logo_x, logo_y), logo)

        print(f"✅ Logo added: {logo_path}")

    elif logo_path:
        # PATH MILA BUT FILE DOESN'T EXIST
        # Favicon download try karo
        print("🔍 Auto-detecting logo...")
        favicon = download_favicon(url)

        if favicon:
            # Same process with favicon
            qr_width, qr_height = img.size
            logo_size = qr_width // 5
            favicon = favicon.resize((logo_size, logo_size), Image.Resampling.LANCZOS)

            circle_size = logo_size + 10
            circle = Image.new("RGBA", (circle_size, circle_size), (0, 0, 0, 0))
            draw = ImageDraw.Draw(circle)
            draw.ellipse((0, 0, circle_size, circle_size), fill=(255, 255, 255, 255))

            circle_x = (qr_width - circle_size) // 2
            circle_y = (qr_height - circle_size) // 2
            logo_x = (qr_width - logo_size) // 2
            logo_y = (qr_height - logo_size) // 2

            img.paste(circle, (circle_x, circle_y), circle)
            img.paste(favicon, (logo_x, logo_y), favicon)
            print("✅ Favicon added as logo!")

    # --- Step 6: Save Karo ---
    img.save(save_as)
    print(f"✅ QR Code saved as: {save_as}")

    return save_as

from generator import generate_qr, auto_detect_logo
# Apne hi project ke functions import!
# generate_qr → QR generate karne ke liye
# auto_detect_logo → Logo auto-detect ke liye

import os
# File existence check ke liye

def main():
    """User-friendly CLI for QR Code Generator."""

    # Step 1: URL lo
    url = input("\n🔗 Enter URL/Link: ").strip()
    if not url:
        print("❌ URL cannot be empty!")
        return  # Empty URL → Function se bahar!

    # Step 2: Logo option
    print("\n🖼️  LOGO OPTIONS:")
    print("1. Auto-detect (URL ke hisaab se)")
    print("2. Custom logo file path")
    print("3. No logo (Simple QR Code)")
    # 3 options — User decide kare!

    logo_choice = input("Choose (1/2/3): ").strip()
    logo_path = None

    if logo_choice == "1":
        logo_path = auto_detect_logo(url)  # Auto-detect
    elif logo_choice == "2":
        logo_path = input("📁 Enter logo file path: ").strip()
        if not os.path.exists(logo_path):
            print(f"❌ File not found: {logo_path}")
            logo_path = None  # Invalid path → No logo
    # Choice 3 → logo_path stays None → No logo!

    # Step 3: Color (Optional)
    custom_color = input("\n🎨 Custom fill color? (Enter for black): ").strip()
    fill_color = custom_color if custom_color else "black"
    # Ternary operator: Value if condition else default

    # Step 4: File name
    default_name = "qr_code.png"
    file_name = input(f"\n💾 File name? (Enter for '{default_name}'): ").strip()

    if not file_name:
        file_name = default_name
    if not file_name.endswith(".png"):
        file_name += ".png"  # Auto .png add!

    # Step 5: Generate!
    print("\n⏳ Generating QR Code...")
    generate_qr(
        url=url,
        logo_path=logo_path,
        fill_color=fill_color,
        save_as=file_name
    )

    print(f"\n🎉 Done! Open '{file_name}' to see your QR Code.")

if __name__ == "__main__":
    main()
# if __name__ == "__main__" → Sirf tab chale jab DIRECT run karo

# ═══════════════════════════════════════════════
# KEY CONCEPTS USED
# ═══════════════════════════════════════════════

# 1. QR Code
#    → Quick Response Code, 2D barcode
#    → Phone se scan → URL open!

# 2. Error Correction (4 Levels)
#    L(7%) → Simple QR
#    M(15%) → Default
#    Q(25%) → Quite High
#    H(30%) → Logo ke saath MUST!

# 3. RGBA Color Mode
#    R, G, B, A = Red, Green, Blue, Alpha
#    Alpha=0 (Transparent) → Alpha=255 (Solid)
#    PNG files support transparency!

# 4. BytesIO
#    Binary data ko memory mein store
#    Disk par file nahi banani → Fast!
#    requests.get() ke response ko Image.open() se read

# 5. urlparse()
#    URL ko parts mein todna
#    "https://github.com/user" → domain = "github.com"

# 6. Floor Division (//)
#    400 // 5 = 80 (Integer, no decimal)
#    Pixel calculations ke liye perfect!

# 7. Center Formula
#    (Parent_Width - Child_Width) // 2
#    Kisi bhi cheez ko center karne ka formula!

# 8. Dictionary for Mapping
#    domain → logo_path
#    If-else chain se better — Scalable!
